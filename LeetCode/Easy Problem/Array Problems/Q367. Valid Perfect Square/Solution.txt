class Solution {
    public boolean isPerfectSquare(int x) {

        int start = 0;
        int end = x;

        while(start <=end){

            int mid = start +(end - start) / 2;

            long sqr = (long) mid* mid;

            if(sqr == x){

                return true;
            }
            
            if(sqr > x){

                end = mid - 1;

            }
            else{

                start = mid+1;
            }
        }
      return false;  
        }
     
    }
